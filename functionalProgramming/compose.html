    # 组合函数

    # 组合函数是什么极其好处：

    # 组合函数将负责不同功能的函数组合起来，完成一个系列的功能；
    # 其好处，负责不同功能的函数式相互解耦没有依赖

    # 组合函数的实现：

    
// 最简单容易理解的，组合两个函数的情况
var compose1 = function(f,g) {
    return function(x) {
        return f(g(x));
    };
};

//写简单一点
const compose2 = (f,g) => (...arg) => f(g(...arg))


    
//组合函数肯定不止两个函数
const compose3 = (...fns) => {
    // 传入的参数，放入输入中就是fns
    return fns.reduce((f, g) => (...args) => g(f(...args)));
};

    # 一个例子

    
function toMath(s) {
    return Number(s);
}

function toFloor(n) {
    return Math.floor(n);
}

function addTwo(s) {
    return s + 2;
}

const handleDeal = compose3(toMath, toFloor, addTwo);
const res = handleDeal('3.412');

console.log('结果是', res); // 5.412
